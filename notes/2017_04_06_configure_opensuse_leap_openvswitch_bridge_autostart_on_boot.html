Configure OpenSUSE Leap OpenvSwitch Bridge Autostart on Boot

ifcfg-br0

cat /etc/sysconfig/network/ifcfg-br0

and the output is: 

TARTMODE='auto'
BOOTPROTO='static'
OVS_BRIDGE='yes'
OVS_BRIDGE_PORT_DEVICE='eth0'
IPADDR='xxx.xxx.xxx.xxx/yy'
BROADCAST=''
ETHTOOL_OPTIONS=''
MTU=''
NAME=''
NETWORK=''
REMOTE_IPADDR=''
STARTMODE='auto'

ifcfg-eth0

cat /etc/sysconfig/network/ifcfg-eth0

and the output is: 

STARTMODE='auto'
BOOTPROTO='none'
