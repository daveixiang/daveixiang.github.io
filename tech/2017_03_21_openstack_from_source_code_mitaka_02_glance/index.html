Install Openstack from source code - glance (mitaka)

2017-03-21
openstack

Create image service database

export GLANCE_DBPASS=openstack
export MYSQL_PASS=openstack

mysql -uroot -p$MYSQL_PASS <<EOF
CREATE DATABASE glance;
GRANT ALL PRIVILEGES ON glance.* TO 'glance'@'localhost' \
  IDENTIFIED BY '$GLANCE_DBPASS';
GRANT ALL PRIVILEGES ON glance.* TO 'glance'@'%' \
  IDENTIFIED BY '$GLANCE_DBPASS';
EOF

Create the image service credentials andAPI endpoints

openstack user create --domain default --password openstack glance
openstack role add --project service --user glance admin

openstack service create --name glance --description "OpenStack Image" image
openstack endpoint create --region RegionOne image public http://controller:9292
openstack endpoint create --region RegionOne image internal http://controller:9292
openstack endpoint create --region RegionOne image admin http://controller:9292

Install glance dependences

yum install $(cat openstack-glance-deps.log | cut -d, -f1 | tr -s '\n'  ' ')

Git clone source code from github

git clone https://github.com/openstack/glance

cd /src/openstack/glance

git checkout stable/mitaka

Build glance source code

cd /src/openstack/glance

python setup.py build

cp -aR etc/* /etc/glance

Make glance configuration

  • glance-api.conf
sed -i -E -e '/^#.+?$/d' -e '/^$/d' /etc/glance/glance-api.conf
ini-config --set /etc/glance/glance-api.conf database connection 'postgresql://glance:openstack@controller/glance'
ini-config --set /etc/glance/glance-api.conf keystone_authtoken auth_uri http://controller:5000
ini-config --set /etc/glance/glance-api.conf keystone_authtoken auth_url http://controller:35357
ini-config --set /etc/glance/glance-api.conf keystone_authtoken auth_type password
ini-config --set /etc/glance/glance-api.conf keystone_authtoken project_domain_name default
ini-config --set /etc/glance/glance-api.conf keystone_authtoken user_domain_name default
ini-config --set /etc/glance/glance-api.conf keystone_authtoken project_name service
ini-config --set /etc/glance/glance-api.conf keystone_authtoken username glance
ini-config --set /etc/glance/glance-api.conf keystone_authtoken password openstack
ini-config --set /etc/glance/glance-api.conf paste_deploy flavor keystone
ini-config --set /etc/glance/glance-api.conf glance_store stores file
ini-config --set /etc/glance/glance-api.conf glance_store default_store file
ini-config --set /etc/glance/glance-api.conf glance_store filesystem_store_datadir '/var/lib/glance/images/'

Create glance-manage, glance-api and glance-registery command

  • glance-manage
cat > glance/cmd/glance-manage <<EOF
#!/bin/python
import sys
from glance.cmd.manage import main
if __name__ == "__main__":
    sys.exit(main())
EOF
  • glance-api
cat > glance/cmd/glance-api <<EOF
#!/bin/python
import sys
from glance.cmd.api import main
if __name__ == "__main__":
    sys.exit(main())
EOF
  • glance-registery
cat > glance/cmd/glance-registry <<EOF
#!/bin/python
import sys
from glance.cmd.registry import main
if __name__ == "__main__":
    sys.exit(main())
EOF

Populate the Image service database and the services

  • populate the image service database
python glance/cmd/glance-manage db_sync
  • start the image services
python glance/cmd/glance-api --config-file=/etc/glance/glance-api.conf --config-file=/etc/glance/glance-api-paste.ini

python glance/cmd/glance-registry --config-file=/etc/glance/glance-registry.conf --config-file=/etc/glance/glance-registry-paste.ini
