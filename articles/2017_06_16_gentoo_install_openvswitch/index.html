Gentoo install openvswitch

2017/06/16

Categories: openvswitch Tags: openvswitch gentoo

Install net-misc/openvswitch

emerge --ask --quiet-build net-misc/openvswitch

Create openvswitch database file

ovsdb-tool create /var/lib/openvswitch/conf.db /usr/share/openvswitch/vswitch.ovsschema

Startup and enable the daemons

rc-update add ovsdb-server default
/etc/init.d/ovsdb-server start

rc-update add ovs-vswitchd default
/etc/init.d/ovs-vswitchd start

Setup auto-configure during system boot

  • configure eth0
cat > /etc/conf.d/net.eth0 <<EOF
config_eth0="null"
EOF

ln -sf /etc/init.d/net.lo /etc/init.d/net.eth0

rc-update add net.eth0

/etc/init.d/net.eth0 start
  • configure br0
cat > /etc/conf.d/net.br0 <<EOF
config_br0="10.0.100.73 netmask 255.255.255.0"
routes_br0="default via 10.0.100.1"
dns_servers_br0="223.5.5.5 223.6.6.6"
EOF

ln -sf /etc/init.d/net.lo /etc/init.d/net.br0

rc-update add net.br0

/etc/init.d/net.br0 start
  • configure openvswitch
cat > /etc/init.d/virt-net.br0  <<EOF
#!/sbin/openrc-run

depend() {
        need ovs-vswitchd
        before net.eth0 net.br0
}

start() {
        ebegin "Enabling virtual switch ports"
                ovs-vsctl --may-exist add-br br0
                ovs-vsctl --may-exist add-port br0 eth0
                ovs-ofctl mod-port br0 br0 up
                ovs-ofctl mod-port br0 eth0 up
        eend \$?
}
EOF

chmod 755 /etc/init.d/virt-net.br0

rc-update add virt-net.br0 default

/etc/init.d/virt-net.br0 start
