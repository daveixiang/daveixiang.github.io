Compile OpenvSwitch 2.5.x LST from source code

2017-01-14
virtualization

Install compile dependencies

Configure and compile the source code

./configure --with-linux=/lib/modules/$(uname -r)/build --prefix=/src/ovs

make -j$(nproc)

make install && make modules_install

Load OpenvSwitch module

modprobe gre
modprobe openvswitch
modprobe libcrc32c

Configure ovsdb

/src/ovs/bin/ovsdb-tool create /src/ovs/etc/openvswitch/conf.db /src/ovs/share/openvswitch/vswitch.ovsschema

Start OVS

Start ovsdb-server

  • without ssl

    /src/ovs/sbin/ovsdb-server --remote=punix:/src/ovs/var/run/openvswitch/db.sock \
--remote=db:Open_vSwitch,Open_vSwitch,manager_options \
--pidfile --detach --log-file

  • with ssl

    /src/ovs/sbin/ovsdb-server --remote=punix:/src/ovs/var/run/openvswitch/db.sock \
--remote=db:Open_vSwitch,Open_vSwitch,manager_options \
--private-key=db:Open_vSwitch,SSL,private_key \
--certificate=db:Open_vSwitch,SSL,certificate \
--bootstrap-ca-cert=db:Open_vSwitch,SSL,ca_cert \
--pidfile --detach --log-file

View /src/ovs/var/log/openvswitch/ovsdb-server.log to get server status

Init OVS

/src/ovs/bin/ovs-vsctl --no-wait init

View /src/ovs/var/log/openvswitch/ovs-vswitchd.log to get ovs-switchd status

Start ovs-switchd

/src/ovs/sbin/ovs-vswitchd --pidfile --detach --log-file
