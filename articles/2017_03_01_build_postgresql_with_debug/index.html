Build PostgreSQL with debug info

2017-03-01
database

Add postgres user

useradd -m -b /src/ postgres

Configure PostgreSQL

./configure --prefix=/src/postgres \
--without-tcl \
--without-gssapi \
--without-pam \
--without-bsd-auth \
--without-ldap \
--without-bonjour \
--without-selinux \
--without-systemd \
--with-uuid=ossp \
--with-libxml \
--with-libxslt \
CFLAGS=' '

Build and install

make -j$(nproc)

su postgres -c 'make install'
