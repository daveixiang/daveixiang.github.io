GCC inline assembly hello world example

2017-02-17
clan
  • x86 example
#include <string.h>
#include <stdio.h>

int main(int argc, char *argv[])
{
        char *str = "Hello World\n";
        long len = strlen(str);
        int ret = 0;
/*
 * ssize_t write(int fd, const void *buf, size_t count);
 *
 */
        __asm__("movl $1, %%eax\n\t"    // write syscall
                "movl $1, %%ebx\n\t"    // fd   
                "movl %1, %%ecx\n\t"    // buf
                "movl %2, %%edx\n\t"    // count
                "syscall\n\t"
                "movl %%eax, %0"        // return result to ret
                : "+m"(ret)
                : "g"(str), "g"(len));
        printf("ret %d\n", ret);

        return 0;
}
