Format ovs-ofctl dump-flows output

2017-06-06
openvswitch
  • Get the openflows of br-int
ovs-ofctl dump-flows   br-int | \
sed -E -e '1d' \
-e 's/(hard_age|idle_age)=[0-9]*?, //g' \
-e 's/ actions/, actions/g' \
-e 's/, /\t/g' | \
awk '{print $3"\t"$6"\t"$7}' | sort -k1,1 -k2,2rV | column -t
