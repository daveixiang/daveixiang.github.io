sheepdog add pthread name

Because sheepdog is multi-thread program. So I think it’s better to add pthread name for convenient.

diff --git a/lib/work.c b/lib/work.c
index 75b0807..3b40bf8 100644
--- a/lib/work.c
+++ b/lib/work.c
@@ -260,11 +260,21 @@ static int create_worker_threads(struct wq_info *wi, size_t nr_threads)
        int ret;

        while (wi->nr_threads < nr_threads) {
+               char thdname[16];
                ret = pthread_create(&thread, NULL, worker_routine, wi);
                if (ret != 0) {
                        sd_err("failed to create worker thread: %m");
                        return -1;
                }
+
+
+               snprintf(thdname, 16, "%s:%zu", wi->name, wi->nr_threads);
+
+               ret = pthread_setname_np(thread, thdname);
+               if (ret != 0) {
+                       sd_err("failed to set  worker thread name : %m");
+               }
+
                wi->nr_threads++;
                sd_debug("create thread %s %zu", wi->name, wi->nr_threads);
        }
